\newcommand\glsform[2][glsdesc]{%
  % Choses the right form of description for acronym
  % \glsv{key}            -> default
  % \glsv[glsuseri]{key}  -> from user1
  % \glsv[Glsuseri]{key}  -> from user1, capitalized
  \ifglsused{#2}%
  {\glsdisp{#2}{\glstext{#2}}}%
  {\glsdisp{#2}{\csname#1\endcsname{#2} (\glstext{#2})}}%
}
